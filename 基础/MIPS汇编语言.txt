C递归过程：
int fact(int n){
	if(n < 1){
		return 1;
	}
	else{
		return n * fact(n-1);
	}
}

MIPS汇编指令格式：
fact:
	addi $sp, $sp, -8
	sw   $ra, 4($sp)
	sw   $a0, 0($sp)
	slti $t0, $a0, 1
	beq  $t0, $zero Ll
	addi $v0, $zero, 1
	addi $sp, $sp, 8
	jr   $ra
	Ll:
		addi $a0, $a0, -1
		jal  fact
		lw   $a0, 0($sp)
		lw   $ra, 4($sp)
		addi $sp, $sp, 8
		mul  $v0, $a0, $v0
		jr   $ra
